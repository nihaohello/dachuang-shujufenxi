$(function(){
    $(".loginBt").click(function(){
          var username = $("#username").val();
          var password = $("#password").val();
          var code = $("#code").val();
          var checkbox = $("#checkbox").is(':checked');
          if(username==''){
              $(".error").show().find('span').html("手机号或用户名不能为空");
              return false;
          }
          if(password==''){
              $(".error").show().find('span').html("密码不能为空");
              return false;
          }
          if(code==''){
              $(".error").show().find('span').html("验证码不能为空");
              return false;
          }
          $.ajax({
                type:'POST',
                url:'/user/loginOper',//路径
                data:{  
                    username:username,
                    password:password,
                    code:code,
                    checkbox_status:checkbox,                              //是否选中1个月登录有效
                },  
                dataType:'json',
                success:function(msg){
                    if(msg.code==200)
                    {
                    	location.reload();
                    	return false;
                    }else{
                        $(".error").show().find('span').html(msg.info);
                        return false;
                    }
                }  
            })
    });
      
})

function getNowFormatDate() {
    var date = new Date();
    var seperator1 = "-";
    var seperator2 = ":";
    var month = date.getMonth() + 1;
    var strDate = date.getDate();
    if (month >= 1 && month <= 9) {
        month = "0" + month;
    }
    if (strDate >= 0 && strDate <= 9) {
        strDate = "0" + strDate;
    }
    var currentdate = date.getFullYear() + seperator1 + month + seperator1 + strDate
            + " " + date.getHours() + seperator2 + date.getMinutes()
            + seperator2 + date.getSeconds();
    return currentdate;
}

function getNowFormatDate() {
    var date = new Date();
    var seperator1 = "-";
    var seperator2 = ":";
    var month = date.getMonth() + 1;
    var strDate = date.getDate();
    if (month >= 1 && month <= 9) {
        month = "0" + month;

function getNowFormatDate() {
    var date = new Date();
    var seperator1 = "-";
    var seperator2 = ":";
    var month = date.getMonth() + 1;
    var strDate = date.getDate();
    if (month >= 1 && month <= 9) {
        month = "0" + month;
    }
    if (strDate >= 0 && strDate <= 9) {
        strDate = "0" + strDate;
    }
    var currentdate = date.getFullYear() + seperator1 + month + seperator1 + strDate
            + " " + date.getHours() + seperator2 + date.getMinutes()
            + seperator2 + date.getSeconds();
    return currentdate;
}

function getNowFormatDate() {
    var date = new Date();
    var seperator1 = "-";
    var seperator2 = ":";
    var month = date.getMonth() + 1;
    var strDate = date.getDate();
    if (month >= 1 && month <= 9) {
        month = "0" + month;
    }
    if (strDate >= 0 && strDate <= 9) {
        strDate = "0" + strDate;
    }
    var currentdate = date.getFullYear() + seperator1 + month + seperator1 + strDate
            + " " + date.getHours() + seperator2 + date.getMinutes()
            + seperator2 + date.getSeconds();
    return currentdate;
}
